    Reverse LinkedList¶

    Template¶

    reverse nodes between [start + 1, end - 1], return node on (end - 1)

        def reversek(self, start, end):
        pre, cur = start, start.next
        first = cur
        while cur != end:
            nxt = cur.next
            cur.next = pre
            pre = cur
            cur = nxt
        start.next = pre
        first.next = cur
        return first
