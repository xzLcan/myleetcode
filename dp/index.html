    Tree DP

    Template

    def dfs(node):
    nonlocal ans
    if not node:
        return 2
    left_state = dfs(node.left)
    right_state = dfs(node.right)
    if left_state == 0 or right_state == 0:
        ans += 1
        return 1
    if left_state == 1 or right_state == 1:
        return 2
    return 0
