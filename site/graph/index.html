    Has Cycle

    Template

    n = len(graph)
status = [0] * n # 0 init, 1, visiting, 2, visited

def has_cycle(node):
    if status[node] == 1:
        return True
    if status[node] == 2:
        return False

    status[node] = 1
    for v in graph[node]:
        if has_cycle(v):
            return True
    status[node] = 2

ans = []
for i in range(n):
    if not has_cycle(i):
        ans.append(i)
return ans

    class UnionFind:
    def __init__(self, size):
        self.parent = list(range(size))  
        self.rank = [1] * size  # height

    def find(self, x):
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x]) 
        return self.parent[x]

    def union(self, x, y):
        rootX = self.find(x)
        rootY = self.find(y)
        if rootX != rootY:
            if self.rank[rootX] > self.rank[rootY]:
                self.parent[rootY] = rootX
            elif self.rank[rootX] < self.rank[rootY]:
                self.parent[rootX] = rootY
            else:
                self.parent[rootY] = rootX
                self.rank[rootX] += 1

    def Dijkstra(self, graph, src):
    dis = [float('inf')] * len(graph)
    dis[src] = 0
    minHeap = [(dis[src], src)]

    while minHeap:
        d, u = heapq.heappop(minHeap)
        if d > dis[u]:
            continue
        for v, w in graph[u]:
            if d + w < dis[v]:
                dis[v] = d + w
                heapq.heappush(minHeap, (dis[v], v))
